// ── Click Event Handler ────────────────── const button = document.querySelector('.trend-card'); button.addEventListener('click', function(event) { // event.target → element that was clicked // event.type → 'click' // event.clientX → X position of click console.log('Clicked at:', event.clientX, event.clientY); event.target.classList.toggle('liked'); }); // Double-click event button.addEventListener('dblclick', (e) => { e.target.style.transform = 'scale(1.05)'; }); // Prevent default + stop propagation link.addEventListener('click', (e) => { e.preventDefault(); e.stopPropagation(); });
// ── Mouse Hover Events ─────────────────── const card = document.querySelector('.trend-card'); // mouseover — fires on element AND children card.addEventListener('mouseover', (e) => { e.target.style.boxShadow = '0 20px 50px rgba(0,0,0,0.15)'; }); // mouseenter — fires only on the element itself card.addEventListener('mouseenter', (e) => { card.classList.add('hovered'); }); // mouseleave — fires when pointer leaves element card.addEventListener('mouseleave', (e) => { card.classList.remove('hovered'); }); // Track relative position within element card.addEventListener('mousemove', (e) => { const rect = card.getBoundingClientRect(); const x = e.clientX - rect.left; const y = e.clientY - rect.top; console.log(`Mouse at: (${x}, ${y})`); });
// ── Keyboard Events ────────────────────── document.addEventListener('keydown', (event) => { console.log(event.key); // 'a', 'Enter', 'ArrowUp'… console.log(event.code); // 'KeyA', 'Enter', 'ArrowUp'… console.log(event.which); // char code (legacy) if (event.shiftKey) { /* Shift held */ } if (event.ctrlKey) { /* Ctrl held */ } if (event.altKey) { /* Alt held */ } // Ctrl+K keyboard shortcut if (event.ctrlKey && event.key === 'k') { openSearch(); } }); document.addEventListener('keyup', (e) => { /* key released */ }); document.addEventListener('keypress', (e) => { /* printable char */ });
// ── Mouse Events Reference ─────────────── const zone = document.getElementById('mouse-canvas'); zone.addEventListener('mousemove', (e) => { const { clientX, clientY, offsetX, offsetY } = e; renderTrail(offsetX, offsetY); }); // Detect which mouse button zone.addEventListener('mousedown', (e) => { switch(e.button) { case 0: handleLeft(); break; case 1: handleMiddle(); break; case 2: handleRight(); break; } }); zone.addEventListener('wheel', (e) => { e.preventDefault(); console.log(e.deltaY); });
// ── Scroll Events ──────────────────────── window.addEventListener('scroll', () => { const scrollY = window.scrollY; const docHeight = document.body.scrollHeight - window.innerHeight; const progress = (scrollY / docHeight) * 100; progressBar.style.width = `${progress}%`; if (scrollY > 50) { navbar.classList.add('scrolled'); } else { navbar.classList.remove('scrolled'); } }); // Intersection Observer (scroll trigger) const observer = new IntersectionObserver((entries) => { entries.forEach((entry) => { if (entry.isIntersecting) { entry.target.classList.add('in-view'); } }); }, { threshold: 0.2 }); document.querySelectorAll('.trend-card') .forEach(el => observer.observe(el));
